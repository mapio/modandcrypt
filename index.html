  return da_posizioni(
    orologio_somma(
      a_posizioni(c), 
      a_posizioni(

  )

def EG(r, n):
  s0, s1, k0, k1 = 1, 0, 0, 1
  while n != 0:
    q, r, n = r // n, n, r % b
    s0, s1 = s1, s0 - q * s1
    k0, k1 = k1, k0 - q * k1
  return r, s0, k0

def expm(b, e, m):
  r = 1
  while e > 0:
    if e % 2 == 1: r = (r * b) % m
    b = (b * b) % m
    e = e // 2
  return r
